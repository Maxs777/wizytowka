function monthName ( month, name ){ let text = ""; switch(month){ case ("grudzień" || "syczeń" || "luty"): text = name + " jeździ na sankach"; break; case ("marzec" || "kwiecień" || "maj") : text = name + " chodzi po kałużach"; break; case ("czerwiec" || "lipiec" || "sierpień") : text = name + " się opala"; break; case ("wrzesień" || "lipiec" || "sierpień") : text = name + " zbiera liście"; break; } return text; }
